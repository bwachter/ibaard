19 // in case we found a certificate we'll set it before negotiating any ssl
20 // stuff. If the other site requests it on handshake SSL will just return
21 // it, if the other site requests it during normal operation SSL will give
22 // us SSL_ERROR_WANT_WRITE or SSL_ERROR_WANT_READ -- which basically means
23 // that OpenSSL takes care about sending the cert _now_ and wants us to
24 // repeat the last operation. In case we did _not_ set a certificate
25 // OpenSSL will call the client_cert_cb()-hook if one is requested. Since
26 // we were unable to load a certifikate on startup we guess we still would
27 // not get it, and use this method just as a simple way to report a
28 // certificate problem

31  (void)cert;
32  (void)pkey;
33  logmsg(L_ERROR, F_SSL, "The server requested a client certificate, but you did not specify one\n", NULL);
34  return 0;
35 }
